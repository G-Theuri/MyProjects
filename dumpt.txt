from bs4 import BeautifulSoup

# The HTML data you provided
data = """
<p><strong>Fabric: </strong>4553-003 Adorn in  Meadow</p>
<p><strong>Grade Level: </strong>D</p>
<p><strong>Pattern Repeat: </strong>H:14, V:13.3</p>
<p><strong>Cutting Direction: </strong>RR</p>
<p><strong>Pattern Match: </strong>Flow Match</p>
<p><strong>Cleaning Code: </strong>W</p>
<p><strong>Contents: </strong>100% Solution-Dyed Acrylic</p>
"""

# Parse the HTML content with BeautifulSoup
soup = BeautifulSoup(data, 'html.parser')

# Create a dictionary to store the parsed data
parsed_data = {}

# Find all <p> tags that contain the <strong> and value we are interested in
for p_tag in soup.find_all('p'):
    # Extract the key (text inside <strong>)
    key_tag = p_tag.find('strong')
    if key_tag:
        key = key_tag.get_text(strip=True).replace(":", "")  # Clean up the key by removing colon and extra spaces
        # Extract the value (text after <strong>)
        value = p_tag.get_text(strip=True).replace(key_tag.get_text(strip=True), "").strip()  # Remove the key part from the text
        parsed_data[key] = value

# Output the parsed data
print(parsed_data)
